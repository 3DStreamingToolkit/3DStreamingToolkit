90 // Beginning of GPU Architecture definitions
91 inline int _ConvertSMVer2CoresDRV(int major, int minor)
92 {
93  // Defines for GPU Architecture types (using the SM version to determine the # of cores per SM
94  typedef struct
95  {
96  int SM; // 0xMm (hexidecimal notation), M = SM Major version, and m = SM minor version
97  int Cores;
98  } sSMtoCores;
99 
100  sSMtoCores nGpuArchCoresPerSM[] =
101  {
102  { 0x10, 8 }, // Tesla Generation (SM 1.0) G80 class
103  { 0x11, 8 }, // Tesla Generation (SM 1.1) G8x class
104  { 0x12, 8 }, // Tesla Generation (SM 1.2) G9x class
105  { 0x13, 8 }, // Tesla Generation (SM 1.3) GT200 class
106  { 0x20, 32 }, // Fermi Generation (SM 2.0) GF100 class
107  { 0x21, 48 }, // Fermi Generation (SM 2.1) GF10x class
108  { 0x30, 192}, // Kepler Generation (SM 3.0) GK10x class
109  { 0x32, 192}, // Kepler Generation (SM 3.2) GK10x class
110  { 0x35, 192}, // Kepler Generation (SM 3.5) GK11x class
111  { 0x37, 192}, // Kepler Generation (SM 3.7) GK21x class
112  { 0x50, 128}, // Maxwell Generation (SM 5.0) GM10x class
113  { 0x52, 128}, // Maxwell Generation (SM 5.2) GM20x class
114  { -1, -1 }
115  };
116 
117  int index = 0;
118 
119  while (nGpuArchCoresPerSM[index].SM != -1)
120  {
121  if (nGpuArchCoresPerSM[index].SM == ((major << 4) + minor))
122  {
123  return nGpuArchCoresPerSM[index].Cores;
124  }
125 
126  index++;
127  }
128 
129  // If we don't find the values, we default use the previous one to run properly
130  printf("MapSMtoCores for SM %d.%d is undefined. Default to use %d Cores/SM\n", major, minor, nGpuArchCoresPerSM[index-1].Cores);
131  return nGpuArchCoresPerSM[index-1].Cores;
132 }

159  dev = 0;
160  }
161 
162  if (dev > deviceCount-1)
163  {
164  fprintf(stderr, "\n");

177

193 inline int gpuGetMaxGflopsDeviceIdDRV()
194 {
195  CUdevice current_device = 0, max_perf_device = 0;
196  int device_count = 0, sm_per_multiproc = 0;
197  int max_compute_perf = 0, best_SM_arch = 0;
198  int major = 0, minor = 0 , multiProcessorCount, clockRate;
199  int devices_prohibited = 0;
200 
201  cuInit(0, __CUDA_API_VERSION, NULL);
202  checkCudaErrors(cuDeviceGetCount(&device_count));
203 
204  if (device_count == 0)
205  {
206  fprintf(stderr, "gpuGetMaxGflopsDeviceIdDRV error: no devices supporting CUDA\n");
207  exit(EXIT_FAILURE);
208  }
209 
210  // Find the best major SM Architecture GPU device
211  while (current_device < device_count)
212  {
213  checkCudaErrors(cuDeviceComputeCapability(&major, &minor, current_device));
214 
215  if (major > 0 && major < 9999)
216  {
217  best_SM_arch = MAX(best_SM_arch, major);
218  }
219 
220  current_device++;
221  }
222 
223  // Find the best CUDA capable GPU device
224  current_device = 0;
225 
226  while (current_device < device_count)
227  {
228  checkCudaErrors(cuDeviceGetAttribute(&multiProcessorCount,
229  CU_DEVICE_ATTRIBUTE_MULTIPROCESSOR_COUNT,
230  current_device));
231  checkCudaErrors(cuDeviceGetAttribute(&clockRate,
232  CU_DEVICE_ATTRIBUTE_CLOCK_RATE,
233  current_device));
234  checkCudaErrors(cuDeviceComputeCapability(&major, &minor, current_device));
235 
236  int computeMode;
237  getCudaAttribute<int>(&computeMode, CU_DEVICE_ATTRIBUTE_COMPUTE_MODE, current_device);
238 
239  if (computeMode != CU_COMPUTEMODE_PROHIBITED)
240  {
241  if (major == 9999 && minor == 9999)
242  {
243  sm_per_multiproc = 1;
244  }
245  else
246  {
247  sm_per_multiproc = _ConvertSMVer2CoresDRV(major, minor);
248  }
249 
250  int compute_perf = multiProcessorCount * sm_per_multiproc * clockRate;
251 
252  if (compute_perf > max_compute_perf)
253  {
254  // If we find GPU with SM major > 2, search only these
255  if (best_SM_arch > 2)
256  {
257  // If our device==dest_SM_arch, choose this, or else pass
258  if (major == best_SM_arch)
259  {
260  max_compute_perf = compute_perf;
261  max_perf_device = current_device;
262  }
263  }
264  else
265  {
266  max_compute_perf = compute_perf;
267  max_perf_device = current_device;
268  }
269  }
270  }
271  else
272  {
273  devices_prohibited++;
274  }
275 
276  ++current_device;
277  }
278 
279  if (devices_prohibited == device_count)
280  {
281  fprintf(stderr, "gpuGetMaxGflopsDeviceIdDRV error: all devices have compute mode prohibited.\n");
282  exit(EXIT_FAILURE);
283  }
284 
285  return max_perf_device;
286 }
