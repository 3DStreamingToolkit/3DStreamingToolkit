18 // NOTE: Do not include any additional headers here. This is an external facing header
19 // so avoid any internal dependencies.
21 // The Unity EventQueue is a standard way of passing information between plugins or parts
22 // of the engine without any reliance on managed systems, game objects or
23 // other pieces of Unity.
24 //
25 // Events are simply a structure and GUID pair. GUIDs were chosen rather
26 // than hashes as it allows plugins to define their own event
27 // payloads and use the EventQueue.
28 //
29 // While the EventQueue itself is self contained and can be used for
30 // many different applications within Unity the global event
31 // queue that is passed to plugins is always serviced on the main
32 // simulation thread and thus serves as a convenient means of
33 // pushing asynchronous operations back to the main game thread
34 //
35 // Payloads should strive to be small. The EventQueue is a 0 allocation
36 // system which ensures that the use of events does not fragment
37 // or add undue pressure on resource limited systems.
38 //
39 // This header defines ALL that is required to listen, define
40 // and send events from both within Unity and from plugins.
42 // USAGE:
43 //
44 // SENDING
45 // ========
46 //
47 // To Define new Event:
48 // --------------------
49 // // Structure and GUID
50 // struct MyEvent {};
51 // REGISTER_EVENT_ID(0x19D736400584B24BULL,0x98B9EFBE26D3F3C9ULL,MyEvent)
52 // // hash above is a GUID split into 2 64 bit integers for convenience.
53 // // DO NOT DUPLICATE GUIDS!
54 //
55 // To Send an Event:
56 // -----------------
57 // // Create Payload and send
58 // MyEvent evt;
59 // GlobalEventQueue::GetInstance ().SendEvent (evt);
60 //
61 // LISTENING
62 // ==========
63 //
64 // STATIC FUNCTION: To Listen For An Event:
65 // ----------------------------------------
66 //
67 // void MyListenFunction (const MyEvent & payload)
68 // { /* do something useful */ }
69 //
70 // // No register that function, create delegate, register delegate with EventQueue
71 // // Lifetime of delegate is important.
72 // StaticFunctionEventHandler<MyEvent> myEventDelegate (&MyListenFunction);
73 // GlobalEventQueue::GetInstance ()->AddHandler (&myEventDelegate);
74 //
75 //
76 // CLASS LISTENER: To Listen on an instance of a class:
77 // ----------------------------------------------------
78 // class MyClass
79 // {
80 // // This method will get called, always called HandleEvent
81 // void HandleEvent(const MyEvent & payload);
82 //
83 // MyClass()
84 // {
85 // // Hookup to listen can be anywhere but here it's in the ctor.
86 // GlobalEventQueue::GetInstance ()->AddHandler (m_myDelegate.SetObject (this));
87 // }
88 //
89 // ~MyClass()
90 // {
91 // // Stop listening.
92 // GlobalEventQueue::GetInstance ()->RemoveHandler (m_myDelegate);
93 // }
94 //
95 // // Delegate is a member of the class.
96 // ClassBasedEventHandler<MyEvent,MyClass> m_myDelegate;
97 // };

155  // EventId - GUID
156  //
157  // To facilitate custom events the EventId object must be a full fledged GUID
158  // to ensure cross plugin uniqueness constraints.
159  //
160  // Template specialization is used to produce a means of looking up an
161  // EventId from it's payload type at compile time. The net result should compile
162  // down to passing around the GUID.
163  //
164  // REGISTER_EVENT_ID should be placed in
165  // the header file of any payload definition OUTSIDE of all namespaces (all EventIds live
166  // in the UnityEventQueue namespace by default) The payload structure and the registration ID are all that
167  // is required to expose the event to other systems.
168  //
169  // There is a tool to generate registration macros for EventIds.
170  struct EventId
171  {
172  public:
173  EventId(unsigned long long high, unsigned long long low)
174  : mGUIDHigh(high)
175  , mGUIDLow(low)
176  {}
177 
178  EventId(const EventId & other)
179  {
180  mGUIDHigh = other.mGUIDHigh;
181  mGUIDLow = other.mGUIDLow;
182  }
183 
184  EventId & operator=(const EventId & other)
185  {
186  mGUIDHigh = other.mGUIDHigh;
187  mGUIDLow = other.mGUIDLow;
188  return *this;
189  }
190 
191  bool Equals(const EventId & other) const { return mGUIDHigh == other.mGUIDHigh && mGUIDLow == other.mGUIDLow; }
192  bool LessThan(const EventId & other) const { return mGUIDHigh < other.mGUIDHigh || (mGUIDHigh == other.mGUIDHigh && mGUIDLow < other.mGUIDLow); }

342  // ============================================================
343  // The Event Queue is a lock free multiple write single read
344  // deferred event system. It uses GUIDs to map payloads to
345  // event handler delegates. The system has some
346  // templates to make registering for events fairly painless
347  // but takes care to try to keep template cost very low.
348  //
349  // NOTE: payloads should be very very small and never allocate
350  // or free memory since they can and will be passed across
351  // dll boundaries.
352  //
353  // There is a hard limit of kMaxEventQueueEventSize bytes for any
354  // payload being passed through this system but payloads that are
355  // this big are probably being handled improperly.
356  UNITY_DECLARE_INTERFACE(IUnityEventQueue)
357  {
358  public:
359  // size of 8196 required for PS4 system events
360  #define kMaxEventQueueEventSize 8196+sizeof(EventId)
361 
362  virtual ~IUnityEventQueue() {
363  }
364 
365  // The primary entry point for firing any
366  // event through the system. The templated
367  // version is simply syntatic sugar for extracting
368  // the EventId from the event object.
369  //
370  // This can be called from any thread.
371  template<typename T>
372  void SendEvent(T & payload)
373  {
374  // Ensure we never fire an event that we can't handle size wise.
375  Assert(sizeof(T) <= (kMaxEventQueueEventSize - sizeof(EventId)));
376 
377  // NOTE: Keep this small to avoid code bloat.
378  // every line of code in here should be scrutinized
379  // as this and GetEventId could easily be sources
380  // of bloat if allowed to grow.
381  SendEventImpl(UnityEventQueue::GetEventId<T>(), (unsigned char*)(&payload), sizeof(T));
382  }

384 #if EVENTQUEUE_SUPPORTS_EVENT_CLEANUP
385  // Some events want to have a destructor.
386  //
387  // You do this by creating a public Destroy()
388  // method on your Event payload. You then
389  // register the event with REGISTER_EVENT_ID_WITH_CLEANUP
390  //
391  // If your event will have someone listening to it right away
392  // then you do not need to do anything else.
393  //
394  // However, if there is a danger that your event will be fired
395  // before someone starts listening for it then you must
396  // call this method somewhere. This will manually register your
397  // class for cleanup. There is no harm in calling this even
398  // if it is not required. If in doubt and your event payload
399  // requires some form of destruction, call this method!
400  template<typename T>
401  void RegisterCleanup(T & payload)
402  {
403  EventHandler * eh = UnityEventQueue::GetEventDestructor<T>();
404  if (eh != NULL)
405  SetCleanupImpl(eh);
406  }

408 #endif
