71  DirectX::XMVECTOR ScreenToVector(float screenx, float screeny)
72  {
73  float x = -( screenx - m_width / 2.f ) / ( m_radius * m_width / 2.f );
74  float y = ( screeny - m_height / 2.f ) / ( m_radius * m_height / 2.f );
75 
76  float z = 0.0f;
77  float mag = x * x + y * y;
78 
79  if( mag > 1.0f )
80  {
81  float scale = 1.0f / sqrtf( mag );
82  x *= scale;
83  y *= scale;
84  }
85  else
86  z = sqrtf( 1.0f - mag );
87 
88  return DirectX::XMVectorSet( x, y, z, 0 );

92  {
93  using namespace DirectX;
94  XMVECTOR dot = XMVector3Dot( vFrom, vTo );

96  return XMVectorSelect( dot, vPart, g_XMSelect1110 );
