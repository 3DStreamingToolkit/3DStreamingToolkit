62 // Device-based resources for holographic cameras are stored in a std::map.
63 // Access this list by providing a callback to this function, and the std::map
64 // will be guarded from add and remove events until the callback returns.
65 // The callback is processed immediately and must not contain any nested calls
66 // to UseHolographicCameraResources.
67 // The callback takes a parameter of type
68 // std::map<UINT32,std::unique_ptr<DX::CameraResources>>& through which the list
69 // of cameras will be accessed.
70 template<typename RetType, typename LCallback>
71 RetType DX::DeviceResources::UseHolographicCameraResources(const LCallback& callback)
72 {
73  std::lock_guard<std::mutex> guard(m_cameraResourcesLock);
74  return callback(m_cameraResources);
